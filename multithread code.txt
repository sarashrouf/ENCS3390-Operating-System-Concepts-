#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <pthread.h>
#include <sys/time.h>

#define MAX_WORDS 17005207
#define MAX_WORD_LENGTH 100
#define NUM_THREADS 4  // You can change the number of threads as needed

typedef struct {
    char word[MAX_WORD_LENGTH];
    int frequency;
} Word;

typedef struct {
    int startIdx;
    int endIdx;
} ThreadData;

pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;  // Mutex for shared memory access

Word *wordArray;
Word *sharedResult;
int wordCount = 0;
int resultSize = 0;

void store_word(Word *wordArray, int *size, const char *word) {
    for (int i = 0; i < *size; i++) {
        if (strcmp(wordArray[i].word, word) == 0) {
            wordArray[i].frequency++;
            return;
        }
    }
    strcpy(wordArray[*size].word, word);
    wordArray[*size].frequency = 1;
    (*size)++;
}

void *process_chunk(void *arg) {
    ThreadData *data = (ThreadData *)arg;
    int startIdx = data->startIdx;
    int endIdx = data->endIdx;
    Word *localArray = malloc(MAX_WORDS * sizeof(Word));
    int localSize = 0;

    for (int i = startIdx; i < endIdx; i++) {
        store_word(localArray, &localSize, wordArray[i].word);
    }

    pthread_mutex_lock(&mutex);  // Lock the shared memory before writing
    for (int i = 0; i < localSize; i++) {
        int found = 0;
        for (int j = 0; j < resultSize; j++) {
            if (strcmp(sharedResult[j].word, localArray[i].word) == 0) {
                sharedResult[j].frequency += localArray[i].frequency;
                found = 1;
                break;
            }
        }
        if (!found) {
            strcpy(sharedResult[resultSize].word, localArray[i].word);
            sharedResult[resultSize].frequency = localArray[i].frequency;
            resultSize++;
        }
    }
    pthread_mutex_unlock(&mutex);  // Unlock after writing

    free(localArray);
    return NULL;
}

void merge(Word *array, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;
    
    Word *leftArray = malloc(n1 * sizeof(Word));
    Word *rightArray = malloc(n2 * sizeof(Word));

    for (int i = 0; i < n1; i++)
        leftArray[i] = array[left + i];
    for (int i = 0; i < n2; i++)
        rightArray[i] = array[mid + 1 + i];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (leftArray[i].frequency >= rightArray[j].frequency) {
            array[k] = leftArray[i];
            i++;
        } else {
            array[k] = rightArray[j];
            j++;
        }
        k++;
    }

    while (i < n1) {
        array[k] = leftArray[i];
        i++;
        k++;
    }
    while (j < n2) {
        array[k] = rightArray[j];
        j++;
        k++;
    }

    free(leftArray);
    free(rightArray);
}

void merge_sort(Word *array, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        merge_sort(array, left, mid);
        merge_sort(array, mid + 1, right);
        merge(array, left, mid, right);
    }
}

int read_file() {
    FILE *data = fopen("text8.txt", "r");
    if (data == NULL) {
        printf("\nAn Error in opening 'text8.txt' file.\n");
        return 0;
    }

    char buffer[MAX_WORD_LENGTH];
    while (fscanf(data, "%s", buffer) != EOF) {
        strcpy(wordArray[wordCount].word, buffer);
        wordArray[wordCount].frequency = 1;  // Dummy frequency for input array
        wordCount++;
    }
    fclose(data);
    return 1;
}

int main() {
    wordArray = malloc(MAX_WORDS * sizeof(Word));
    if (wordArray == NULL) {
        printf("Memory allocation failed for wordArray.\n");
        return 1;
    }

    if (!read_file()) {
        free(wordArray);
        return 1;
    }

    sharedResult = malloc(MAX_WORDS * sizeof(Word));
    if (sharedResult == NULL) {
        printf("Memory allocation failed for sharedResult.\n");
        free(wordArray);
        return 1;
    }

    int numThreads = NUM_THREADS;
    int segmentSize = wordCount / numThreads;
    pthread_t threads[numThreads];
    ThreadData threadData[numThreads];

    struct timeval start, end;
    gettimeofday(&start, NULL);

    // Create threads
    for (int i = 0; i < numThreads; i++) {
        threadData[i].startIdx = i * segmentSize;
        threadData[i].endIdx = (i == numThreads - 1) ? wordCount : threadData[i].startIdx + segmentSize;
        pthread_create(&threads[i], NULL, process_chunk, &threadData[i]);
    }

    // Join threads
    for (int i = 0; i < numThreads; i++) {
        pthread_join(threads[i], NULL);
    }

    // Sort the results using merge sort
    merge_sort(sharedResult, 0, resultSize - 1);

    gettimeofday(&end, NULL);
    double time_taken = (end.tv_sec - start.tv_sec) + (end.tv_usec - start.tv_usec) / 1e6;

    printf("Top 10 most frequent words:\n");
    for (int i = 0; i < 10 && i < resultSize; i++) {
        printf("%s: %d\n", sharedResult[i].word, sharedResult[i].frequency);
    }
    printf("\nExecution Time: %.6f seconds\n", time_taken);

    // Free resources
    free(wordArray);
    free(sharedResult);

    return 0;
}
